addingAvailableBikes: (
      state,
      action: PayloadAction<{
        pickUpDate: string;
        pickUpTime: string;
        dropOffDate: string;
        dropOffTime: string;
      }>
    ) => {
      const { pickUpDate, pickUpTime, dropOffDate, dropOffTime } =
        action.payload;
      const requestedStart = new Date(pickUpDate + "T" + pickUpTime).getTime();
      const requestedEnd = new Date(dropOffDate + "T" + dropOffTime).getTime();
      console.log("requestedStart", requestedStart);
      console.log("requestedEnd", requestedEnd);
      state.availableBikes = state.all.filter(
        (vehicle) =>
          // pickUpDate: "2025-09-03",
          // dropOffDate: "2025-09-06",
          // pickUpTime: "08:00",
          // dropOffTime:
          !vehicle.bookedSlots.some((slot) => {
            const slotStart = new Date(
              slot.pickUpDate + "T" + slot.pickUpTime
            ).getTime();
            const slotEnd = new Date(
              slot.dropOffDate + "T" + slot.dropOffTime
            ).getTime();
            console.log("expectedStart", slotStart);
            console.log("expectedEnd", slotEnd);
            return requestedStart < slotEnd && slotStart < requestedEnd;
          })
      );
    },
